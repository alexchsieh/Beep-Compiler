function add
(
	var int a;
	var int b;
)
{
  return a + b;
}


func mult
(
	var int a,
	var int b
)
{
  return a * b;
}



func main()
{
  var int a;
  var int b;
  var int c;
  var int d;


  a = 100;
  b = 50;
  c = add(a, b);
  write c;        // should print 150


  d = mult(c, a + b);
  write d;        // should print "22500", since 22500 = 150 * 150;
}
